CREATE TABLE `kategooriad` (
  `id` int(11) NOT NULL AUTO_INCREMENT,
  `Nimi` varchar(45) CHARACTER SET utf8 COLLATE utf8_general_ci NOT NULL,
  `Tag` varchar(45) DEFAULT NULL,
  PRIMARY KEY (`id`),
  UNIQUE KEY `id_UNIQUE` (`id`),
  UNIQUE KEY `Nimi_UNIQUE` (`Nimi`)
) ENGINE=InnoDB AUTO_INCREMENT=4 DEFAULT CHARSET=utf8



CREATE TABLE `tooted` (
  `id` int(11) NOT NULL AUTO_INCREMENT,
  `Nimi` varchar(45) DEFAULT NULL,
  `kategooriaId` int(11) NOT NULL,
  `Tag` varchar(45) DEFAULT NULL,
  `Kirjeldus` varchar(45) DEFAULT NULL,
  `Pilt` varchar(45) DEFAULT NULL,
  `Hind` decimal(10,0) unsigned zerofill DEFAULT NULL,
  `Kogus` int(10) unsigned zerofill DEFAULT NULL,
  `kategooria_id` int(11) NOT NULL,
  PRIMARY KEY (`id`),
  UNIQUE KEY `id_UNIQUE` (`id`),
  KEY `fk_kategooriad_2_tooted_idx` (`kategooriaId`),
  CONSTRAINT `fk_kategooriad_2_tooted` FOREIGN KEY (`kategooriaId`) REFERENCES `kategooriad` (`id`)
) ENGINE=InnoDB AUTO_INCREMENT=15 DEFAULT CHARSET=utf8
